# Cloudflare Workers — Git connect build settings
# Use these when importing code-by-jerry/fellowmart in the dashboard.

# Repository: code-by-jerry/fellowmart

# Build settings (Workers & Pages → Create → Import Git):

#   Deploy command:             npm run deploy

#   NEXT_PUBLIC_APP_URL          (production URL, not localhost)
#   NEXT_PUBLIC_SITE_URL         (same as APP_URL)
#   SUPABASE_SERVICE_ROLE_KEY
#   IMAGEKIT_PUBLIC_KEY
#   IMAGEKIT_PRIVATE_KEY
#   IMAGEKIT_URL_ENDPOINT
#   RAZORPAY_KEY_ID
#   RAZORPAY_KEY_SECRET
#   RAZORPAY_WEBHOOK_SECRET
#   PLATFORM_ADMIN_EMAIL

#   Repo → Settings → Secrets and variables → Actions
#   Add the same keys plus:
#     CLOUDFLARE_API_TOKEN
#     CLOUDFLARE_ACCOUNT_ID
#   Push to master triggers .github/workflows/deploy-cloudflare.yml
